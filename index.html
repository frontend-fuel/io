#define AOUT_PIN A1 // A1 pin 
#define LED 9 
int value; 
void Soil_Moisture_Sensor(); 
void setup()  
{ 
  Serial.begin(9600); 
  pinMode(LED, OUTPUT); 
} 
void loop()  
{ 
  Soil_Moisture_Sensor();  
} 
void Soil_Moisture_Sensor() 
{ 
  value = analogRead(AOUT_PIN); // read the analog value from sensor 
  Serial.print("Moisture value: "); 
  Serial.println(value); 
  float moisturePercent = map(value, 1023, 300, 0, 100);
  moisturePercent = constrain(moisturePercent, 0, 100);
  Serial.print("Moisture Percentage: ");
  Serial.print(moisturePercent);
  Serial.println("%");
  if (moisturePercent < 30) 
  { 
    digitalWrite(LED, HIGH); 
    Serial.println("⚠️ WARNING: Soil moisture below 30% - TOO DRY!"); 
  } 
  else 
  { 
    digitalWrite(LED, LOW); 
    Serial.println("Soil moisture is sufficient."); 
  } 
  delay(500); 
}

#include <Adafruit_GFX.h>
#include <Adafruit_ST7735.h>
#include <DHT.h>

#define TFT_CS 10
#define TFT_RST 9
#define TFT_DC 8
#define DHTPIN 2
#define Buzzer 4
#define DHTTYPE DHT11

DHT dht(DHTPIN, DHTTYPE);
Adafruit_ST7735 tft = Adafruit_ST7735(TFT_CS, TFT_DC, TFT_RST);

void setup() {
  Serial.begin(9600);
  pinMode(Buzzer, OUTPUT);
  dht.begin();
  tft.initR(INITR_BLACKTAB);
  tft.setRotation(3);
}

void loop() {
  float temp = dht.readTemperature();
  float hum = dht.readHumidity();

  if (!isnan(temp) && !isnan(hum)) {
    // Print to Serial Monitor
    Serial.print("Temp: "); Serial.print(temp); Serial.print(" C ");
    Serial.print("Hum: "); Serial.println(hum);

    // Display on TFT
    tft.fillScreen(ST77XX_BLACK);
    tft.setTextColor(ST77XX_WHITE);
    tft.setTextSize(1);
    tft.setCursor(10, 10);
    tft.print("Temp: "); tft.print(temp); tft.print(" C");
    tft.setCursor(10, 30);
    tft.print("Hum: "); tft.print(hum); tft.print(" %");

    // Display alert if needed
    if (temp > 35 || hum < 30) {
      tft.setTextColor(ST77XX_RED);
      tft.setCursor(10, 50);
      tft.print("ALERT!");
    }

    // Control Buzzer based on temperature
    digitalWrite(Buzzer, temp > 29 ? HIGH : LOW);
  }

  delay(2000);
}

#define TRIG 9
#define ECHO 10
#define BUZZER 6
#define G 4
#define B 5
#define R 3

void setup() {
  Serial.begin(9600);
  pinMode(TRIG, OUTPUT);
  pinMode(ECHO, INPUT);
  pinMode(BUZZER, OUTPUT);
  pinMode(G, OUTPUT);
  pinMode(B, OUTPUT); 
  pinMode(R, OUTPUT);
}

void loop() {
  digitalWrite(TRIG, LOW); 
  delayMicroseconds(2);
  digitalWrite(TRIG, HIGH); 
  delayMicroseconds(10); 
  digitalWrite(TRIG, LOW);
  float d = pulseIn(ECHO, HIGH) * 0.034 / 2;
  Serial.print("Distance: "); 
  Serial.print(d); 
  Serial.println(" cm");

  if (d > 10) {
    digitalWrite(G, HIGH);  // Green LED ON - Slot Available
    digitalWrite(B, LOW);
    digitalWrite(R, LOW);
    noTone(BUZZER);
  }
  else if (d > 5) {
    digitalWrite(G, LOW);
    digitalWrite(B, HIGH);
    digitalWrite(R, LOW);
    tone(BUZZER, 1000, 200);
    delay(500);
    noTone(BUZZER);
  }
  else {
    // Vehicle detected very close (within 10 cm)
    digitalWrite(G, LOW);  // Indicating slot is occupied
    digitalWrite(B, LOW);
    digitalWrite(R, HIGH);
    tone(BUZZER, 1000);      // Continuous alert if needed
    Serial.println("Slot Occupied");
  }

  delay(200);
}

from machine import ADC, Pin
import time

water_sensor = ADC(Pin(26))
buzzer = Pin(15, Pin.OUT)

while True:
    raw_value = water_sensor.read_u16()
    voltage = (raw_value / 65535) * 3.3
    percentage = (raw_value / 65535) * 100

    print(f"Raw: {raw_value}, Voltage: {voltage:.2f}V, Water Level: {percentage:.2f}%")

    if percentage > 80:
        buzzer.value(1)
    else:
        buzzer.value(0)

    time.sleep(1)

#include <Wire.h>
#include <BH1750.h>

BH1750 lightMeter;

void setup() {
  Serial.begin(9600);
  Wire.begin();
  pinMode(7, OUTPUT);

  if (!lightMeter.begin(BH1750::CONTINUOUS_HIGH_RES_MODE)) {
    Serial.println("BH1750 Error");
    while (1);
  }

  Serial.println("BH1750 Ready");
}

void loop() {
  float lux = lightMeter.readLightLevel();
  Serial.print("Light: "); Serial.print(lux); Serial.println(" lx");

  digitalWrite(7, lux < 90 ? HIGH : LOW);
  Serial.println(lux < 90 ? "Street Light ON" : "Street Light OFF");

  delay(1000);
}

#include <SoftwareSerial.h>

#define LED 13
#define R 5
#define G 6
#define B 7

SoftwareSerial bt(2, 3); // RX, TX

void setup() {
  Serial.begin(9600);
  bt.begin(9600);
  pinMode(LED, OUTPUT);
  pinMode(R, OUTPUT);
  pinMode(G, OUTPUT);
  pinMode(B, OUTPUT);
}

void loop() {
  if (bt.available()) {
    char cmd = bt.read();
    
    // Clear the serial buffer before printing the received command.
    if (cmd == 'O') {
      digitalWrite(LED, HIGH);
      Serial.println("LED ON");
    }
    else if (cmd == 'F') {
      digitalWrite(LED, LOW);
      digitalWrite(R, LOW);
      digitalWrite(G, LOW);
      digitalWrite(B, LOW);
      Serial.println("LED OFF, RGB OFF");
    }
    else if (cmd == 'R') {
      digitalWrite(R, HIGH);
      digitalWrite(G, LOW);
      digitalWrite(B, LOW);
      Serial.println("Red LED ON");
    }
    else if (cmd == 'G') {
      digitalWrite(R, LOW);
      digitalWrite(G, HIGH);
      digitalWrite(B, LOW);
      Serial.println("Green LED ON");
    }
    else if (cmd == 'B') {
      digitalWrite(R, LOW);
      digitalWrite(G, LOW);
      digitalWrite(B, HIGH);
      Serial.println("Blue LED ON");
    }
  }
}
